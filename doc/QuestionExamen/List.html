 def lpush(other)
   if @head == nil
    @head = other
    @tail = other
  else
    other.thenext = @head
    @head.theprev = other
    other.theprev = nil
    @head = other 
  end
end

def reverse_each #recorre la lista en orden inverso
    aux = @tail
    while(aux) do
            yield aux
            aux = aux.theprev
    end
end
